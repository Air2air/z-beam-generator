def _build_prompt(
        self,
        material_name: str,
        material_data: Dict,
        author_info: Optional[Dict] = None,
        frontmatter_data: Optional[Dict] = None,
        schema_fields: Optional[Dict] = None,
    ) -> str:
        """
        Build prompt using both material data and frontmatter data.
        """
        return f"""Generate YAML frontmatter metatags for {material_name} laser cleaning webpage using the provided frontmatter data.
Your output should be STRICTLY in YAML format with proper frontmatter delimiters (---).
Do NOT generate an HTML page or any explanatory text - ONLY the YAML frontmatter with metadata.
Follow the exact structure of the example file with title, meta_tags list, opengraph list, twitter list, canonical URL, and alternate list.
Include the material's name in the title and all relevant fields."""
